-- 创建一个包含多种数据类型的表来测试数据库对不同类型的支持
CREATE TABLE test_types (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100),
    value DECIMAL(10, 2),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    is_active BOOLEAN
);

-- 添加索引来测试查询优化和性能
CREATE INDEX idx_test_name ON test_types (name);

-- 创建一个带有外键约束的表来测试关系完整性
CREATE TABLE related_table (
    id SERIAL PRIMARY KEY,
    type_id INT REFERENCES test_types(id),
    description TEXT
);
